def solve_and_plot(potential_func,N=2000,dx=1.0/2000):
    """
    
    
    
    potential_func:,numpynumpy
    N:
    dx:
    
    """
    # x
    x=np.linspace(0,1,N+1)
    V=potential_func(x)
    print(f'x:{x}')
    print(f'x:{len(x)}')
    
    # 
    d=1/dx**2+potential_func(x)[1:-1]
    e=-1/(2*dx**2)*np.ones(len(d)-1)
    
    # 
    w,v=eigh_tridiagonal(d,e)
    
    # 
    # 
    plt.figure(figsize=(10, 4))
    plt.plot(x, V, label='Potential V(x)')
    plt.title('Potential')
    plt.xlabel('x')
    plt.ylabel('V(x)')
    plt.show()
    
     # 
    plt.figure(figsize=(10, 4))
    for i in range(4):  # 
        plt.plot(v.T[i], label=f'{i}th state')
    plt.title('Wavefunctions')
    plt.xlabel('dx')
    plt.ylabel('psi(x)')
    plt.legend()
    plt.show()
    
    # 
    plt.figure(figsize=(10, 4))
    for i in range(4):  # 
        plt.plot(v.T[i]**2, label=f'{i}th state')
    plt.title('Probability densities')
    plt.xlabel('dx')
    plt.ylabel('|psi(x)|^2')
    plt.legend()
    plt.show()
    
      # 
    plt.figure(figsize=(10, 4))
    plt.bar(np.arange(0,10,1), w[:10])
    plt.title('Eigenvalues')
    plt.xlabel('n')
    plt.ylabel('Energy')
    # 
    for i in range(10):
        plt.axhline(w[i], color='black', linestyle='--', linewidth=0.5)
        plt.text(i-0.5, w[i], f'{w[i]:.2f}', fontsize=8)
    # 
    for i in range(9):
        print(f'{i}{w[i+1]-w[i]}')
    plt.show()
    # 
    A=np.zeros((10,10))
    for i in range(10):
        for j in range(10):
            A[i,j]=sum(v[:,i]*v[:,j])
    plt.imshow(A)
    plt.colorbar()
    plt.title('Orthogonality')
    plt.xlabel('i')
    plt.ylabel('j')
    plt.show()
